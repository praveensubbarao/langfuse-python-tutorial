Configure LLM-as-a-judge in the UI
set up hallucination + completeness evaluators

LLM-as-a-judge evaluators live entirely in the Langfuse UI — no code required to set them up. Once configured, they fire automatically on every new trace that matches your filter.

Step-by-step setup:

1
Go to Langfuse UI → Evaluation → Evaluators → Create
2
Choose LLM-as-a-Judge type. Pick a built-in template or write your own rubric.
3
Set the target: Observations (scores each LLM call) or Traces (scores the whole pipeline).
4
Choose score type: Numeric (0–1 for gradients), Boolean (pass/fail), Categorical (labels).
5
Enable Fast Mode toggle — required for the evaluator to fire on live production traces.
6
Save and Enable. Send a new trace from your chatbot — the evaluator fires within seconds.

Create these two evaluators now — they cover the most critical QA Architect chatbot quality signals:

RUBRIC Evaluator 1: answer_completeness (NUMERIC 0–1)
Name: answer_completeness
Type: LLM-as-a-Judge
Score type: NUMERIC (min: 0, max: 1)
Target: Observations (type = generation)
Fast Mode: ON

Judge prompt:
---
You are evaluating a QA Architect chatbot response for completeness.

User question: {{input}}
Assistant response: {{output}}

Score the response from 0.0 to 1.0:
- 1.0: Fully answers the question with definition, context, and practical guidance
- 0.7: Answers the main question but misses some important aspects
- 0.4: Partially answers but leaves key parts unaddressed
- 0.0: Does not answer the question at all

Return ONLY a JSON object: {"score": 0.85, "reasoning": "one sentence explanation"}
---
RUBRIC Evaluator 2: is_actionable (BOOLEAN)
Name: is_actionable
Type: LLM-as-a-Judge
Score type: BOOLEAN
Target: Observations (type = generation)
Fast Mode: ON

Judge prompt:
---
You are evaluating a QA Architect chatbot response.

User question: {{input}}
Assistant response: {{output}}

Does the response give actionable, practical advice the user can implement immediately?
- TRUE (1): Response contains specific steps, examples, or concrete techniques
- FALSE (0): Response is vague, theoretical only, or doesn't answer what was asked

Return ONLY: {"score": 1, "reasoning": "one sentence"}
---
After saving both evaluators: Run your 05_prompt_chatbot.py from Day 2 H1 and ask 3-4 questions. Within 30–60 seconds, Langfuse will score each generation with answer_completeness and is_actionable automatically. You'll see the scores appear on the trace without touching any Python code.

Custom LLM judge via Python SDK
01_custom_judge.py — run your own judge manually

Sometimes you need a judge that runs in your own pipeline — not the automated UI evaluator. This pattern lets you run a judge model against any trace and push the score back to Langfuse.

PYTHON day2/01_custom_judge.py
from dotenv import load_dotenv
import os, json
load_dotenv("../.env")

from langfuse import get_client, observe, propagate_attributes
from openai import OpenAI

langfuse = get_client()
client   = OpenAI(base_url=os.getenv("OLLAMA_BASE_URL"), api_key="ollama")

JUDGE_PROMPT = """You are an expert evaluator of QA Architect chatbot responses.

Question: {question}
Response: {response}

Evaluate on THREE criteria. Return ONLY valid JSON, nothing else:
{{
  "helpfulness": {{
    "score": 0.0_to_1.0,
    "reasoning": "one sentence"
  }},
  "technical_accuracy": {{
    "score": 0.0_to_1.0,
    "reasoning": "one sentence"
  }},
  "has_example": {{
    "score": 1_or_0,
    "reasoning": "one sentence"
  }}
}}"""


def run_judge(question: str, response: str, trace_id: str, observation_id: str = None):
    """Run the judge model and push scores back to Langfuse."""
    judge_response = client.chat.completions.create(
        model=os.getenv("OLLAMA_MODEL"),
        messages=[{
            "role": "user",
            "content": JUDGE_PROMPT.format(question=question, response=response)
        }],
        temperature=0.1,
    )

    raw = judge_response.choices[0].message.content.strip()

    try:
        # Strip markdown fences if model adds them
        clean = raw.replace("```json", "").replace("```", "").strip()
        scores = json.loads(clean)
    except json.JSONDecodeError:
        print(f"  [WARN] Judge returned non-JSON: {raw[:100]}")
        return

    # Push all three scores to Langfuse
    for name, result in scores.items():
        try:
            score_val = float(result["score"])
            data_type = "BOOLEAN" if name == "has_example" else "NUMERIC"

            langfuse.create_score(
                name=f"judge_{name}",
                value=score_val,
                data_type=data_type,
                trace_id=trace_id,
                observation_id=observation_id,
                comment=result.get("reasoning", ""),
            )
            print(f"  ✓ {name}: {score_val} — {result.get('reasoning','')[:60]}")
        except (KeyError, ValueError) as e:
            print(f"  [WARN] Score parse error for {name}: {e}")


@observe(name="qa-generation", as_type="generation",
         capture_input=False, capture_output=False)
def get_qa_answer(question: str) -> tuple[str, str]:
    """Returns (answer, observation_id)"""
    system = "You are a QA Architect. Give detailed, practical answers with examples."
    langfuse.update_current_generation(
        model=os.getenv("OLLAMA_MODEL"),
        input=[{"role": "user", "content": question}],
    )
    response = client.chat.completions.create(
        model=os.getenv("OLLAMA_MODEL"),
        messages=[
            {"role": "system", "content": system},
            {"role": "user",   "content": question},
        ],
    )
    answer = response.choices[0].message.content
    obs_id = langfuse.get_current_observation_id()
    langfuse.update_current_generation(output=answer,
        usage={"input": response.usage.prompt_tokens, "output": response.usage.completion_tokens})
    return answer, obs_id


if __name__ == "__main__":
    questions = [
        "What is risk-based testing and how do you prioritise test cases?",
        "Explain the difference between unit, integration, and E2E tests.",
        "How do you measure test coverage effectively?",
    ]

    for q in questions:
        trace_id = None
        obs_id   = None

        with propagate_attributes(
            trace_name=f"Judge eval: {q[:40]}",
            user_id="user-123",
            tags=["judge-eval", "day2"],
            metadata={"tutorial": "day2-hour3"},
        ):
            answer, obs_id = get_qa_answer(q)
            trace_id = langfuse.get_current_trace_id()

        print(f"\nQ: {q}")
        print(f"A: {answer[:120]}...")
        print("Running judge...")
        run_judge(q, answer, trace_id, obs_id)
        print()

    langfuse.flush()
    print("✓ All traces judged. Check Langfuse → Traces for judge_* scores.")

JSON parsing note: Ollama models sometimes wrap JSON in markdown code fences. The code above strips these — always clean judge output before parsing. If you see persistent parse errors, add format="json" to the Ollama API call (supported by llama3.2 and mistral).

Code evaluators — deterministic checks
02_code_evals.py — zero-cost, zero-LLM quality gates

Code evaluators run pure Python logic against your traces. They're free (no LLM tokens), instant, and 100% deterministic. Use them for everything that doesn't need semantic judgment.

Code evaluators in Langfuse UI (Launch Week #5 feature): You can also write Python code evaluators directly in the Langfuse UI under Evaluation → Code Evaluators. They run server-side on every matching observation. The pattern below shows the equivalent logic you'd run in your own pipeline — same checks, your infrastructure.
PYTHON day2/02_code_evals.py
from dotenv import load_dotenv
import os, json, re
load_dotenv("../.env")

from langfuse import get_client, observe, propagate_attributes
from openai import OpenAI

langfuse = get_client()
client   = OpenAI(base_url=os.getenv("OLLAMA_BASE_URL"), api_key="ollama")


# ── Deterministic evaluators — no LLM needed ─────────────────────

def eval_min_length(text: str, min_words: int = 20) -> tuple[int, str]:
    """BOOLEAN: response has at least min_words words."""
    count = len(text.split())
    passed = count >= min_words
    return (1 if passed else 0,
            f"{count} words ({'pass' if passed else 'fail'}, min={min_words})")


def eval_no_apology(text: str) -> tuple[int, str]:
    """BOOLEAN: response doesn't start with excessive apologies."""
    apology_starts = ["i'm sorry", "i apologize", "unfortunately i cannot"]
    first_50 = text.lower()[:50]
    found = any(a in first_50 for a in apology_starts)
    return (0 if found else 1, "apology detected" if found else "no apology")


def eval_keyword_coverage(answer: str, question: str) -> tuple[float, str]:
    """NUMERIC: fraction of question nouns found in answer."""
    stopwords = {"what","is","the","a","an","how","do","you","and","of","to"}
    q_words = {w.lower() for w in re.findall(r'\b\w{4,}\b', question)} - stopwords
    if not q_words:
        return 1.0, "no keywords to check"
    answer_lower = answer.lower()
    found = sum(1 for w in q_words if w in answer_lower)
    score = round(found / len(q_words), 3)
    return score, f"{found}/{len(q_words)} keywords covered"


def eval_no_hallucinated_urls(text: str) -> tuple[int, str]:
    """BOOLEAN: response doesn't contain invented URLs."""
    urls = re.findall(r'https?://\S+', text)
    return (0 if urls else 1,
            f"Found URLs: {urls[:2]}" if urls else "no URLs")


def run_code_evals(question: str, answer: str, trace_id: str, obs_id: str = None):
    """Run all deterministic evals and push scores to Langfuse."""
    evals = [
        ("min_length_20w",         "BOOLEAN",  eval_min_length(answer, 20)),
        ("no_apology_start",       "BOOLEAN",  eval_no_apology(answer)),
        ("keyword_coverage",       "NUMERIC",  eval_keyword_coverage(answer, question)),
        ("no_hallucinated_urls",   "BOOLEAN",  eval_no_hallucinated_urls(answer)),
    ]

    for name, dtype, (value, comment) in evals:
        langfuse.create_score(
            name=name, value=float(value),
            data_type=dtype, comment=comment,
            trace_id=trace_id, observation_id=obs_id,
        )
        status = "✓" if (float(value) > 0) else "✗"
        print(f"  {status} {name}: {value} ({comment})")


@observe(name="code-eval-qa", as_type="generation",
         capture_input=False, capture_output=False)
def get_answer(question: str) -> tuple[str, str]:
    response = client.chat.completions.create(
        model=os.getenv("OLLAMA_MODEL"),
        messages=[
            {"role": "system", "content": "You are a QA Architect. Be thorough."},
            {"role": "user",   "content": question},
        ],
    )
    answer = response.choices[0].message.content
    obs_id = langfuse.get_current_observation_id()
    langfuse.update_current_generation(
        model=os.getenv("OLLAMA_MODEL"),
        input=[{"role": "user", "content": question}],
        output=answer,
        usage={"input": response.usage.prompt_tokens, "output": response.usage.completion_tokens},
    )
    return answer, obs_id


if __name__ == "__main__":
    questions = [
        "What is exploratory testing?",
        "Hi",   # ← too short — will fail min_length
        "How do you build a regression test suite for a REST API?",
    ]

    for q in questions:
        trace_id = obs_id = None
        with propagate_attributes(
            trace_name=f"Code-eval: {q[:40]}",
            user_id="user-123", tags=["code-evals", "day2"],
            metadata={"tutorial": "day2-hour3"},
        ):
            answer, obs_id = get_answer(q)
            trace_id = langfuse.get_current_trace_id()

        print(f"\nQ: {q}")
        print(f"A: {answer[:80]}...")
        print("Code evals:")
        run_code_evals(q, answer, trace_id, obs_id)

    langfuse.flush()
