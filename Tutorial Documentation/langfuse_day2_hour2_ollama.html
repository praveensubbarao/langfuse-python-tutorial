Score types & the data model
four types · three attachment levels

A score in Langfuse is a structured evaluation result attached to a trace or observation. Think of it as the answer to "how good was this response?" — expressed as a number, a label, a yes/no, or a text note.

NUMERIC
Continuous quality measure
Any float value. Great for helpfulness (0.0–1.0), relevance scores, similarity, response length penalties.
value=0.85 → "helpfulness"
BOOLEAN
True/false binary decision
Value must be 0 or 1 (not True/False). Perfect for thumbs up/down, correct/incorrect, on-topic/off-topic.
value=1 → "user_thumbs_up"
CATEGORICAL
Discrete label from a set
String value from a predefined set. Good for quality tiers (excellent/good/poor), response type labels, intent classification.
value="good" → "response_quality"
TEXT
Free-form reviewer notes
String up to 500 chars. Good for human reviewer comments, bug descriptions, qualitative notes alongside a numeric score.
value="Answer was correct but verbose"

Scores can be attached at three levels:

Score attachment levels
SESSION ← score an entire multi-turn conversation
TRACE ← score one user request end-to-end (most common)
OBSERVATION ← score a specific span or generation inside the trace
score name="helpfulness" value=0.9 data_type="NUMERIC"
score name="user_feedback" value=1 data_type="BOOLEAN"
Rule of thumb: Attach scores to traces for overall quality (user feedback, end-to-end correctness). Attach to observations (specific generations) when you want to evaluate the LLM call itself — faithfulness, hallucination, response format. You'll use both today.
2
Three ways to create scores
choose the right method for each situation
METHOD 1
Low-level create_score()
Pass trace_id + observation_id explicitly. Works from anywhere — inside or outside a span context, even in a different process.
Use for: async feedback, post-processing pipelines, CI evaluators
METHOD 2
span.score() / span.score_trace()
Score the current span object directly. You hold a reference to the span and call score on it inline.
Use for: scores computed right after the operation completes
METHOD 3
score_current_span/trace()
Context-aware — scores whichever span/trace is currently active in the OTel context. No span reference needed.
Use for: scores computed inside @observe functions where you don't hold the span object
MethodRequires trace_id?Inside @observe?Async-safe?
create_score()✓ Yes — pass explicitlyAny time✓ Yes
span.score()Auto from spanHold span ref✓ Yes
score_current_span/trace()Auto from contextMust be inside with/observe✓ Yes

Numeric scores
01_numeric_scores.py — quality floats on traces

Numeric scores are the most flexible type. You compute a float and attach it to a trace or observation. Common examples: response length, readability, similarity to a reference answer.

PYTHON day2/01_numeric_scores.py
from dotenv import load_dotenv
import os
load_dotenv("../.env")

from langfuse import get_client, observe, propagate_attributes
from openai import OpenAI

langfuse = get_client()
client   = OpenAI(base_url=os.getenv("OLLAMA_BASE_URL"), api_key="ollama")


def compute_length_score(text: str, target_words: int = 100) -> float:
    """Score 0-1: how close the response length is to the target."""
    word_count = len(text.split())
    ratio = word_count / target_words
    # Perfect score at target, decreases for too short or too long
    return round(max(0.0, 1.0 - abs(1.0 - ratio)), 3)


def compute_structure_score(text: str) -> float:
    """Score 0-1: does the response use structured formatting?"""
    indicators = ["1.", "2.", "•", "-", "\n", "first", "second"]
    hits = sum(1 for i in indicators if i in text.lower())
    return round(min(1.0, hits / 3), 3)


@observe(name="qa-answer-scored", as_type="generation",
         capture_input=False, capture_output=False)
def ask_and_score(question: str) -> str:
    system = "You are a QA Architect. Answer in structured bullet points, 80-120 words."

    langfuse.update_current_generation(
        model=os.getenv("OLLAMA_MODEL"),
        input=[
            {"role": "system", "content": system},
            {"role": "user",   "content": question},
        ],
    )

    response = client.chat.completions.create(
        model=os.getenv("OLLAMA_MODEL"),
        messages=[
            {"role": "system", "content": system},
            {"role": "user",   "content": question},
        ],
    )
    answer = response.choices[0].message.content
    langfuse.update_current_generation(output=answer,
        usage={"input": response.usage.prompt_tokens, "output": response.usage.completion_tokens})

    # ── METHOD 3: score_current_span/trace inside @observe ───────

    # Score the generation (observation-level)
    langfuse.score_current_span(
        name="response_length",
        value=compute_length_score(answer, target_words=100),
        data_type="NUMERIC",
        comment=f"{len(answer.split())} words (target: 100)",
    )

    # Score the overall trace
    langfuse.score_current_trace(
        name="structure_score",
        value=compute_structure_score(answer),
        data_type="NUMERIC",
        comment="Measures use of numbered lists, bullets, sections",
    )

    return answer

        "What is regression testing?",

        "What makes a good test case?",

            trace_name="Numeric-scored QA",
            user_id="user-123",
            tags=["scoring", "numeric", "day2"],

            answer = ask_and_score(q)

Boolean scores — thumbs up/down
02_user_feedback.py — the most common feedback pattern

Boolean scores (value 0 or 1) are the standard pattern for user feedback widgets — the thumbs up/down you see in almost every chat interface. The critical detail: value must be a float 0 or 1, not Python's True/False.

v4 gotcha: data_type="BOOLEAN" is required — otherwise Langfuse infers value=0 or value=1 as NUMERIC. You must pass the data_type explicitly for boolean scores.
PYTHON day2/02_user_feedback.py
from dotenv import load_dotenv
import os, uuid
load_dotenv("../.env")

from langfuse import get_client, observe, propagate_attributes
from openai import OpenAI

langfuse = get_client()
client   = OpenAI(base_url=os.getenv("OLLAMA_BASE_URL"), api_key="ollama")


@observe(name="qa-response", as_type="generation",
         capture_input=False, capture_output=False)
def get_answer(question: str) -> tuple[str, str]:
    """Returns (answer, trace_id) so we can score async after user sees it."""
    system = "You are a QA Architect. Be concise."
    langfuse.update_current_generation(
        model=os.getenv("OLLAMA_MODEL"),
        input=[{"role": "user", "content": question}],
    )
    response = client.chat.completions.create(
        model=os.getenv("OLLAMA_MODEL"),
        messages=[
            {"role": "system", "content": system},
            {"role": "user",   "content": question},
        ],
    )
    answer = response.choices[0].message.content
    langfuse.update_current_generation(output=answer,
        usage={"input": response.usage.prompt_tokens, "output": response.usage.completion_tokens})
    return answer


def submit_user_feedback(trace_id: str, thumbs_up: bool, comment: str = ""):
    """
    Called AFTER the user sees the response and clicks thumbs up/down.
    This is METHOD 1: create_score() with explicit trace_id.
    Runs outside the original span context — perfectly valid.
    """
    langfuse.create_score(
        name="user_feedback",
        value=1 if thumbs_up else 0,   # ← float 0 or 1, NOT True/False
        data_type="BOOLEAN",              # ← required for boolean
        trace_id=trace_id,
        comment=comment if comment else None,
    )
    emoji = "👍" if thumbs_up else "👎"
    print(f"  {emoji} Feedback recorded for trace {trace_id[:12]}...")


def interactive_feedback_session():
    """
    Simulate a real chatbot UI:
    1. User asks a question
    2. Bot answers
    3. User gives thumbs up/down
    4. Score is recorded against the trace
    """
    session_id = f"feedback-{uuid.uuid4().hex[:8]}"
    print(f"\n🤖 QA Architect — Feedback Demo")
    print(f"   Session: {session_id}")
    print(f"   After each answer, rate with 👍 (y) or 👎 (n)")
    print(f"   Type 'quit' to exit\n")

    while True:
        question = input("You: ").strip()
        if question.lower() in ("quit", "q"):
            break
        if not question:
            continue

        # Capture the trace_id so we can score it after user sees the answer
        trace_id = None

        with propagate_attributes(
            trace_name=f"Feedback: {question[:40]}",
            session_id=session_id,
            user_id="user-123",
            tags=["feedback-demo", "day2"],
            metadata={"tutorial": "day2-hour2"},
        ):
            answer = get_answer(question)
            # Get the current trace ID while still inside propagate_attributes
            trace_id = langfuse.get_current_trace_id()

        print(f"\nQA Architect: {answer}\n")

        # Ask for feedback OUTSIDE the trace context (realistic — async)
        rating = input("Was this helpful? (y/n/skip): ").strip().lower()
        if rating in ("y", "n") and trace_id:
            comment = input("Optional comment (or Enter to skip): ").strip()
            submit_user_feedback(
                trace_id=trace_id,
                thumbs_up=(rating == "y"),
                comment=comment,
            )
        print()

    langfuse.flush()
    print(f"\n✓ Session complete. Check Sessions → {session_id}")


if __name__ == "__main__":
    interactive_feedback_session()

Categorical scores
03_categorical_scores.py — quality tiers and labels

Categorical scores assign discrete labels to traces. They're ideal when you want named quality tiers (excellent/good/poor) or classification labels (on-topic/off-topic, safe/unsafe). Values must be strings.

PYTHON day2/03_categorical_scores.py
from dotenv import load_dotenv
import os
load_dotenv("../.env")

from langfuse import get_client, observe, propagate_attributes
from openai import OpenAI

langfuse = get_client()
client   = OpenAI(base_url=os.getenv("OLLAMA_BASE_URL"), api_key="ollama")

# Category definitions — deterministic rules (no LLM needed)
def classify_response_quality(answer: str) -> str:
    """Simple heuristic quality classification."""
    words = len(answer.split())
    has_structure = any(c in answer for c in ["1.", "•", "\n-", "First"])
    has_example   = any(w in answer.lower() for w in ["example", "e.g.", "for instance"])

    if words >= 60 and has_structure and has_example:
        return "excellent"
    elif words >= 30 and (has_structure or has_example):
        return "good"
    elif words >= 15:
        return "acceptable"
    else:
        return "poor"


def classify_topic(question: str) -> str:
    """Classify the QA topic for routing/analytics."""
    q = question.lower()
    if any(w in q for w in ["regression", "smoke", "sanity", "test type"]):
        return "test-types"
    elif any(w in q for w in ["automation", "selenium", "pytest", "framework"]):
        return "automation"
    elif any(w in q for w in ["agile", "sprint", "scrum", "kanban"]):
        return "agile-qa"
    elif any(w in q for w in ["performance", "load", "stress"]):
        return "performance"
    else:
        return "general"


@observe(name="qa-classified", as_type="generation",
         capture_input=False, capture_output=False)
def answer_and_classify(question: str) -> str:
    response = client.chat.completions.create(
        model=os.getenv("OLLAMA_MODEL"),
        messages=[
            {"role": "system", "content": "You are a QA Architect. Give structured answers with examples."},
            {"role": "user",   "content": question},
        ],
    )
    answer = response.choices[0].message.content
    langfuse.update_current_generation(
        model=os.getenv("OLLAMA_MODEL"),
        input=[{"role": "user", "content": question}],
        output=answer,
        usage={"input": response.usage.prompt_tokens, "output": response.usage.completion_tokens},
    )

    quality  = classify_response_quality(answer)
    topic    = classify_topic(question)

    # Categorical score on the generation (observation-level)
    langfuse.score_current_span(
        name="response_quality",
        value=quality,           # ← string value e.g. "excellent"
        data_type="CATEGORICAL",
        comment=f"words={len(answer.split())}",
    )

    # Categorical score on the trace (trace-level)
    langfuse.score_current_trace(
        name="question_topic",
        value=topic,             # ← string value e.g. "test-types"
        data_type="CATEGORICAL",
    )

    print(f"  Quality: {quality} | Topic: {topic}")
    return answer


if __name__ == "__main__":
    questions = [
        "What is regression testing and how often should you run it?",
        "How do you integrate automated tests into a CI/CD pipeline?",
        "How should a QA team work in an Agile sprint?",
        "What is performance testing?",
        "Hi",   # ← short answer — expect "poor" quality score
    ]

    for q in questions:
        with propagate_attributes(
            trace_name=f"Classified: {q[:35]}",
            user_id="user-123",
            tags=["categorical", "day2"],
            metadata={"tutorial": "day2-hour2"},
        ):
            answer = answer_and_classify(q)
            print(f"Q: {q[:60]}")
            print(f"A: {answer[:80]}...\n")

    langfuse.flush()
    print("✓ In Langfuse → Traces, filter by 'response_quality = poor' to find bad responses")

Observation-level scores
scoring a specific span, not just the trace

Sometimes you want to score a specific step inside a trace — not the whole thing. For example: score the retrieval span on how relevant the docs were, independently of the LLM answer quality.

PYTHON observation_scoring pattern
# METHOD 1: create_score with both trace_id AND observation_id
with langfuse.start_as_current_observation(
    as_type="span", name="retrieval"
) as retrieval_span:

    docs = do_retrieval(query)

    # Score the retrieval span directly via span.score()
    retrieval_span.score(
        name="retrieval_relevance",
        value=compute_relevance(docs, query),
        data_type="NUMERIC",
        comment=f"{len(docs)} chunks retrieved",
    )

    # Also score the parent trace from within the span
    retrieval_span.score_trace(
        name="retrieval_succeeded",
        value=1 if docs else 0,
        data_type="BOOLEAN",
    )


# METHOD 1 (explicit): pass observation_id to create_score()
# Useful when scoring AFTER the span has closed
langfuse.create_score(
    name="retrieval_relevance",
    value=0.72,
    data_type="NUMERIC",
    trace_id=the_trace_id,
    observation_id=retrieval_span.id,  # pins score to that specific span
    comment="Post-hoc relevance score",
)
When observation_id matters: In a trace with 10 spans, all scores without observation_id appear on the trace root. If you want to see which retrieval call was weak (not which overall trace was weak), always pass observation_id. In the UI, these scores appear on the individual span card in the timeline, not just the trace header.

Async user feedback — the real-world pattern
score arrives minutes after the trace closes

In production, user feedback never arrives instantly. A user reads the answer, thinks about it, then maybe clicks 👎 three minutes later. The trace is long since closed. This is the most important pattern to understand.

PYTHON async feedback endpoint pattern
# Simulates a FastAPI endpoint that receives user feedback
# The trace was created minutes ago in a different request

# In your LLM endpoint: generate answer AND return trace_id to client
def answer_question(question: str) -> dict:
    trace_id = None

    with propagate_attributes(
        trace_name="user-query",
        user_id="user-123",
    ):
        answer = call_llm(question)
        trace_id = langfuse.get_current_trace_id()

    langfuse.flush()   # ensure trace is sent before returning

    # Return BOTH the answer and the trace_id to the frontend
    return {"answer": answer, "trace_id": trace_id}


# In your feedback endpoint: user clicks 👎 on the frontend
# Frontend sends back the trace_id it received with the answer
def receive_feedback(trace_id: str, helpful: bool, comment: str = ""):
    langfuse.create_score(
        name="user_feedback",
        value=1 if helpful else 0,
        data_type="BOOLEAN",
        trace_id=trace_id,   # ← links to the existing trace
        comment=comment or None,
    )
    langfuse.flush()

# ✅ Langfuse accepts scores for traces that were created in the past.
# The score will appear on the trace even if hours have passed.
Frontend integration tip: In a React/Vue frontend, store the trace_id in local state alongside the answer. When the user clicks 👍/👎, POST to your backend with {"trace_id": "...", "helpful": true}. Your backend calls langfuse.create_score(). The Langfuse dashboard immediately reflects the feedback — no polling needed.

Score configs — enforce schema on scores
define valid ranges & categories in the UI

Score configs define what's valid for a given score name. Without a config, any float is accepted for NUMERIC and any string for CATEGORICAL. With a config, Langfuse validates values and enables the annotation UI for human reviewers.

Config fieldNUMERICCATEGORICALBOOLEAN
Range validation min / max (e.g. 0–1 for normalised scores) N/A Always 0 or 1
Categories N/A Exact strings (e.g. "excellent", "good", "poor") N/A
Enables UI annotation ✓ Yes — slider in UI ✓ Yes — dropdown in UI ✓ Yes — toggle in UI

Create configs in the Langfuse UI: Settings → Scores → Create Score Config. For now, just know they exist — you'll use them heavily in Hour 3 when setting up LLM-as-a-judge evaluators.

Categorical scores without a config: Any string is accepted — even typos. A config enforces the valid set (e.g. only "excellent", "good", "acceptable", "poor"). If you're building a team annotation workflow, always create a config first.

Hour 2 mini-project: scored feedback chatbot
04_scored_chatbot.py — all score types in one chatbot

Combine everything: the Day 2 H1 prompt-managed chatbot now gets automatic quality scores on every response, and asks the user for a thumbs up/down after each answer. Low-quality traces get flagged in the dashboard automatically.

PYTHON day2/04_scored_chatbot.py
from dotenv import load_dotenv
import os, uuid
load_dotenv("../.env")

from langfuse import observe, get_client, propagate_attributes
from openai import OpenAI

langfuse = get_client()
client   = OpenAI(base_url=os.getenv("OLLAMA_BASE_URL"), api_key="ollama")
FALLBACK  = "You are an experienced QA Architect. Provide practical advice."


# ── Scoring helpers ──────────────────────────────────────────────
def _length_score(text: str) -> float:
    words = len(text.split())
    return round(max(0.0, 1.0 - abs(1.0 - words / 80)), 3)

def _quality_tier(text: str) -> str:
    words = len(text.split())
    structured = any(c in text for c in ["1.", "•", "- ", "\n\n"])
    has_example = "example" in text.lower() or "e.g." in text.lower()
    if   words >= 60 and structured and has_example: return "excellent"
    elif words >= 30 and (structured or has_example):   return "good"
    elif words >= 10:                                      return "acceptable"
    else:                                                   return "poor"


@observe(name="llm-call", as_type="generation",
         capture_input=False, capture_output=False)
def call_llm(messages: list, prompt_obj, turn: int) -> str:
    langfuse.update_current_generation(
        model=os.getenv("OLLAMA_MODEL"),
        input=messages, prompt=prompt_obj,
        metadata={"turn": str(turn)},
    )
    response = client.chat.completions.create(
        model=os.getenv("OLLAMA_MODEL"),
        messages=messages,
    )
    reply = response.choices[0].message.content
    langfuse.update_current_generation(
        output=reply,
        usage={"input": response.usage.prompt_tokens, "output": response.usage.completion_tokens},
    )

    # Auto-score the generation immediately after the LLM call
    langfuse.score_current_span(
        name="response_length_score",
        value=_length_score(reply),
        data_type="NUMERIC",
        comment=f"{len(reply.split())} words",
    )
    langfuse.score_current_trace(
        name="auto_quality",
        value=_quality_tier(reply),
        data_type="CATEGORICAL",
    )
    return reply


@observe(name="chat-turn")
def process_turn(messages: list, prompt_obj, turn: int) -> str:
    return call_llm(messages, prompt_obj, turn)


class ScoredChatbot:
    def __init__(self, user_id="user-123"):
        self.user_id    = user_id
        self.session_id = f"scored-{uuid.uuid4().hex[:8]}"
        self.turn       = 0

        try:
            self.prompt_obj = langfuse.get_prompt(
                "qa-architect-system", cache_ttl_seconds=300, fallback=FALLBACK
            )
            system = self.prompt_obj.compile(
                experience_years="20",
                specialisation="quality strategy",
                style="concise with examples",
            )
        except:
            self.prompt_obj, system = None, FALLBACK

        self.history = [{"role": "system", "content": system}]
        v = getattr(self.prompt_obj, "version", "fallback")
        print(f"\n{'='*52}\n  Scored QA Chatbot — Prompt v{v} · Session {self.session_id}\n  y=👍  n=👎  skip=no rating  quit=exit\n{'='*52}\n")

    def chat(self, user_input: str) -> tuple[str, str]:
        self.turn += 1
        self.history.append({"role": "user", "content": user_input})

        trace_id = None
        with propagate_attributes(
            trace_name=f"Turn {self.turn}: {user_input[:40]}",
            session_id=self.session_id, user_id=self.user_id,
            tags=["scored", "day2"],
            metadata={"turn": str(self.turn)},
        ):
            reply    = process_turn(self.history, self.prompt_obj, self.turn)
            trace_id = langfuse.get_current_trace_id()

        self.history.append({"role": "assistant", "content": reply})
        return reply, trace_id

    def record_feedback(self, trace_id: str, helpful: bool, comment: str = ""):
        langfuse.create_score(
            name="user_feedback",
            value=1 if helpful else 0,
            data_type="BOOLEAN",
            trace_id=trace_id,
            comment=comment or None,
        )

    def end(self):
        langfuse.flush()
        print(f"\n✓ {self.turn} turns. Session: {self.session_id}")
        print(f"  Dashboard → Sessions → filter scores: auto_quality='poor' | user_feedback=false")


if __name__ == "__main__":
    bot = ScoredChatbot(user_id="user-123")
    while True:
        q = input("You: ").strip()
        if q.lower() in ("quit", "q"): break
        if not q: continue

        reply, trace_id = bot.chat(q)
        print(f"\nQA Architect: {reply}\n")

        rating = input("Helpful? (y/n/skip): ").strip().lower()
        if rating in ("y", "n") and trace_id:
            comment = input("Comment (Enter to skip): ").strip()
            bot.record_feedback(trace_id, rating == "y", comment)
            print(f"  {'👍' if rating == 'y' else '👎'} recorded\n")

    bot.end()
